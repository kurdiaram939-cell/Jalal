local isSearched = false
-- دروستکردنی گۆڕدراوێک وەک گڵۆپ بۆ ئەوەی بزانین گەڕان کراوە یان نا (سەرەتا کوژاوەتەوە).

local config = {
-- دروستکردنی خشتەیەکی سەرەکی بۆ پاشەکەوتکردنی زانیاری و کۆدی هێکسی بەردەکان.

    [1] = {name="زمروتی سەوز", hex={"6D656708h", "00000032h", "00000000h", "00000000h", "00000000h", "00000000h"}},
    -- خانەی یەکەم: ناوی بەردی یەکەم و ئەو ٦ کۆدە هێکسەی کە دەبێت بنوسرێنە ناو میمۆری.

    [2] = {name="یاقوتی زەرد", hex={"6D656708h", "00000031h", "00000000h", "00000000h", "00000000h", "00000000h"}},
    -- خانەی دووەم: ناوی بەردی دووەم و کۆدە هێکسە تایبەتەکانی خۆی.

    [3] = {name="یاقوتی سور", hex={"6D656708h", "00000033h", "00000000h", "00000000h", "00000000h", "00000000h"}}
    -- خانەی سێیەم: ناوی بەردی سێیەم و کۆدە هێکسە تایبەتەکانی خۆی.

}
-- داخستنی خشتەی سەرەکی زانیارییەکان.

function menuYaqut()
-- دەستپێکردنی دروستکردنی فەنکشنێک بە ناوی menuYaqut بۆ کۆکردنەوەی کۆدەکانی مێنوو و هاکەکە.

    local menu = gg.multiChoice({
-- دروستکردنی مێنوویەکی فرە-هەڵبژاردن (تیک لێدان) لە ناو گەیم گاردندا.

        "1 🟢 زمروتی سەوز",
        -- بژاردەی یەکەم لە مێنووەکەدا بۆ تیک لێدان.

        "2 🟡 یاقوتی زەرد",
        -- بژاردەی دووەم لە مێنووەکەدا بۆ تیک لێدان.

        "3 🔴 یاقوتی سور",
        -- بژاردەی سێیەم لە مێنووەکەدا بۆ تیک لێدان.

        "⬅️ گەڕانەوە"
        -- بژاردەی چوارەم لە مێنووەکەدا بۆ دەرچوون یان گەڕانەوە.

    }, nil, "✨ SEROK ARAM YAQUT ✨")
    -- دیاریکردنی ناونیشانی سەرەکی مێنووەکە لە سەرەوە.

    if menu == nil then return end
    -- مەرجێک: ئەگەر بەکارهێنەر مێنووەکەی داخست یان Cancelی کرد، فەنکشنەکە ڕابگرە و هیچ مەکە.

    if menu[4] then gg.clearResults() gg.clearList() os.exit() end
    -- مەرجێک: ئەگەر کلیک لەسەر بژاردەی ٤ (گەڕانەوە) کرا، میمۆری پاک بکەرەوە و بە تەواوی لە ئەسکریپتەکە دەرچۆ.

    -- پشکنین: ئایا هیچ قاڵبێک هەڵبژێردراوە؟
    -- کۆمێنتێکە بۆ ڕوونکردنەوەی ئەوەی کە کۆدی دێڕەکانی خوارەوە چی دەکەن.

    local selected = false
    -- دروستکردنی گۆڕدراوێکی کاتی بۆ پشکنینی ئەوەی کە ئایا تیک لە بەردەکان دراوە یان نا (سەرەتا نەخێرە).

    for i=1, 3 do if menu[i] then selected = true break end end
    -- لۆپێک لە ١ تا ٣ دەگەڕێت، ئەگەر هەر خانەیەکیان تیکی لێدرابوو، گۆڕدراوەکە دەکاتە True و لۆپەکە دەوەستێنێت.

    if not selected then return end
    -- مەرجێک: ئەگەر بەکارهێنەرەکە تیکی لە چوارگۆشەی هیچ بەردێک نەدابوو، فەنکشنەکە ڕابگرە و مەڕۆ بۆ قۆناغی گەڕان.

    -- گەڕان تەنها بۆ یەکەم جار
    -- کۆمێنتێکە بۆ ڕوونکردنەوەی ئەوەی کە ئەم بەشە تەنها یەکجار گەڕان دەکات.

    if not isSearched then
-- مەرجێک: ئەگەر گڵۆپی گەڕانەکە کوژابۆوە (واتە جاری یەکەم بێت)، بچۆ ناو ئەم بلۆکە کۆدە.

        gg.clearResults()
        -- سڕینەوەی هەموو ئەنجام و گەڕانە کۆنەکانی ناو میمۆری گەیم گاردن.

        gg.searchNumber("65537~65542;1970225964;29", 3)
        -- گەڕانی گرووپی بەدوای ئەم سێ ژمارەیە لە میمۆریدا، ژمارە ٣ واتە بە جۆری گەڕانی XOR.

        gg.refineNumber("29", 3)
        -- پاڵاوتنی ئەنجامەکان؛ تەنها ئەو ئەنجامانە دەهێڵێتەوە کە بەهاکەیان بە تەواوی یەکسانە بە ٢٩.

        isSearched = true
        -- گڵۆپی گەڕانەکە دادەکێشێت و دەیکاتە True بۆ ئەوەی جاری داهاتوو ئەم گەڕانە کاتگرە دووبارە نەبێتەوە.

    end
    -- کۆتایی مەرجی گەڕانی یەکەمجار.

    local count = gg.getResultCount()
    -- ژمارەی ئەو ئەنجامانەی لە میمۆریدا دۆزراونەتەوە دەژمێرێت و دەیخاتە ناو لۆکاڵی count.

    if count == 0 then 
-- مەرجێک: ئەگەر ژمارەی ئەنجامەکان یەکسان بوو بە سفر (واتە هیچی نەدۆزییەوە):

        isSearched = false 
        -- گڵۆپی گەڕانەکە دەکوژێنێتەوە بۆ ئەوەی جاری داهاتوو دووبارە بگەڕێتەوە.

        gg.alert("هیچ ئەنجامێک نەدۆزرایەوە! تکایە بچۆ شوێنی گەڕانە ڕاستەکە.") 
        -- نامەیەکی ئاگادارکردنەوە پیشانی بەکارهێنەر دەدات کە کێشەیەک هەیە.

        return 
        -- فەنکشنەکە ڕادەگرێت و ناچێتە قۆناغی گۆڕین.

    end
    -- کۆتایی مەرجی نەدۆزینەوەی ئەنجام.

    local res = gg.getResults(count)
    -- هەموو ناونیشانە (Address) دۆزراوەکانی ناو میمۆری دەخوێنێتەوە و دەیانخاتە ناو خشتەیەک بە ناوی res.

    local input = gg.prompt({'بڕی پێویست بنوسە:'}, {'100'}, {'number'})
    -- پیشاندانی سندوقی نووسین بۆ بەکارهێنەر بۆ دیاریکردنی بڕی بەردەکە، بەهای سەرەتایی ١٠٠ـە و تەنها ژمارە وەردەگرێت.

    if input == nil then return end
    -- مەرجێک: ئەگەر بەکارهێنەر لە سندوقی نووسینەکەدا Cancelی کرد، پرۆسەکە لێرەدا ڕابگرە.

    gg.clearList()
    -- پاککردنەوەی لیستی پارێزراوی (Saved List) ناو گەیم گاردن.

    local edit = {}
    -- دروستکردنی خشتەیەکی بەتاڵ بە ناوی edit بۆ کۆکردنەوەی ئەو بەهایانەی کە بڕیارە بگۆڕدرێن.

    local freeze = {}
    -- دروستکردنی خشتەیەکی بەتاڵ بە ناوی freeze بۆ کۆکردنەوەی ئەو بەهایانەی کە پێویستە جێگیر ببن.

    for i, v in pairs(config) do
-- لۆپێک بە ناو خشتەی سەرەکی زانیاری بەردەکاندا (config) دەگەڕێت.

        if menu[i] then
-- مەرجێک: ئەگەر بەکارهێنەرەکە لە مێنووەکەدا تیکی لە بەردی ژمارە i دابوو:

            for _, r in ipairs(res) do
-- لۆپێکی ناوەکی، یەک بە یەک بە ناو تەواوی ئەو ناونیشانانەدا دەگەڕێت کە لە گەڕانەکەدا دۆزرابوونەوە.

                local check = gg.getValues({{address = r.address + 12, flags = 4}})[1].value
                -- ناونیشانی ئەسڵی وەردەگرێت، ١٢ باێت دەچێتە پێشەوە (Offset 12) و بەهای ئەو شوێنە بە جۆری DWORD دەخوێنێتەوە.

                if check == 1 or check == 2 then
-- مەرجێک: تەنها ئەگەر بەهای ئەو ئۆفسێتە یەکسان بوو بە ١ یان ٢، ڕێگا بە گۆڕینی کۆدەکە دەدات.

                    table.insert(freeze, {address = r.address + 12, value = 2, flags = 4, freeze = true})
                    -- ناونیشانی ئۆفسێتی ١٢ ئامادە دەکات بۆ فریزکردن (جێگیرکردن) لەسەر بەهای ٢ بۆ ئەوەی یارییەکە نەیگۆڕێتەوە.

                    for j, h in ipairs(v.hex) do 
-- لۆپێک بە ناو ئەو ٦ کۆدە هێکسەی بەردە هەڵبژێردراوەکەدا دەگەڕێت.

                        table.insert(edit, {address = r.address + 12 + (j * 4), value = h, flags = 4}) 
                        -- کۆدە هێکسەکان ڕیز دەکات بۆ گۆڕین، ٤ باێت ٤ باێت دەچێتە پێشەوە بۆ ئەوەی هێکسەکان بە ڕێکوپێکی لە دوای یەک دابنێت.

                    end
                    -- کۆتایی لۆپی ڕیزکردنی کۆدە هێکسەکان.

                    table.insert(edit, {address = r.address + 40, value = 0, flags = 4})
                    -- ناونیشانی ئۆفسێتی ٤٠ ئامادە دەکات بۆ گۆڕین و بەهاکەی دەکات بە 0.

                    table.insert(edit, {address = r.address + 44, value = input[1], flags = 4})
                    -- ناونیشانی ئۆفسێتی ٤٤ ئامادە دەکات و ئەو بڕە ژمارەیەی تێدا دەنوسێت کە بەکارهێنەرەکە لە سندوقی نووسینەکەدا نووسیبووی.

                end
                -- کۆتایی مەرجی پشکنینی ئۆفسێتی ١٢ (ئەگەر ١ یان ٢ بوو).

            end
            -- کۆتایی لۆپی گەڕان بە ناو ناونیشانە دۆزراوەکاندا.

        end
        -- کۆتایی مەرجی تیک لێدانی بەردەکە لە مێنوودا.

    end 
    -- کۆتایی لۆپی گەڕان بە ناو خشتەی configـدا.

    if #edit > 0 then
-- مەرجێک: ئەگەر خشتەی edit بەتاڵ نەبوو و ناونیشانی تێدا بوو بۆ گۆڕین:

        gg.setValues(edit) 
        -- ناردن و نووسینی هەموو کۆدە هێکسەکان و بڕی نوێی بەردەکان بە یەکجار بۆ ناو میمۆری یارییەکە (هاکەکە لێرەدا ڕوودەدات).

        gg.addListItems(freeze)
        -- ناردنی ئەو ناونیشانانەی کە مەرجی فریزکردنیان هەبوو بۆ ناو لیستی پارێزراوی گەیم گاردن تا جێگیر ببن.

        gg.toast("بە سەرکەوتوویی جێگیر کرا ✅")
        -- پەیامێکی خێرا لە خوارەوەی شاشەکە پیشان دەدات کە کارەکە بە سەرکەوتوویی جێبەجێ بوو.

    else
-- ئەگەر خشتەی edit بەتاڵ بوو (واتە مەرجی ئۆفسێتی ١٢ لە چاودێری میمۆرییەکەدا ڕاست نەبوو):

        gg.alert("هیچ گۆڕانکارییەک نەکرا! (خانەی ٣ مەرجەکەی تێدا نییە)")
        -- نامەیەکی ئاگادارکردنەوە پیشان دەدات کە هاکەکە جێبەجێ نەکراوە چونکە مەرجەکەی تێدا نەبووە.

    end
    -- کۆتایی مەرجی پشکنینی خشتەی edit.

end
-- کۆتایی تەواوی فەنکشنی menuYaqut.

-- گرنگترین بەش: لووپەکە بۆ ئەوەی سکرێپتەکە کۆتایی نەیەت
-- کۆمێنتێکە بۆ ڕوونکردنەوەی ئەوەی کە کۆدی خوارەوە لۆپی سەرەکی ئەسکریپتەکەیە.

while true do 
-- دەستپێکردنی لۆپێکی ئەبەدی و بێبڕانەوە بۆ ئەوەی ئەسکریپتەکە هەمیشە لە باکگراونددا بە کراوەیی بمێنێتەوە.

    if gg.isVisible() then 
-- مەرجێک: هەر کاتێک بەکارهێنەر کلیکی لەسەر ئایکۆنی لێڵ یان سەرەکی گەیم گاردین کرد لە ناو یارییەکەدا:

        gg.setVisible(false) 
        -- پەنجەرە گەورەکەی گەیم گاردن دەشارێتەوە بۆ ئەوەی شاشەی یارییەکە تێک نەچێت.

       menuYaqut()
       -- فەنکشنی مێنووی بەردەکان کە لە سەرەتاوە دروستمان کردبوو بانگ دەکاتەوە تا ڕاستەوخۆ مێنووی تیک لێدانەکە پیشان بداتەوە.

    end 
    -- کۆتایی مەرجی بینینی گەیم گاردین.

    gg.sleep(100) 
    -- ئەسکریپتەکە بۆ ماوەی ١٠٠ میلی چرکە دەخەوێنێت لە نێوان هەر دووبارەبوونەوەیەکدا بۆ ئەوەی سی پی یووی مۆبایلەکە ماندوو نەبێت و یارییەکە لاگ نەکات.

end
-- کۆتایی لۆپە بێبڕانەوەکە (while).
